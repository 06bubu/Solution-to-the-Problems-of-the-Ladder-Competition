```cpp []
void solve() {
    int n,k,m;
    cin>>n>>m>>k;
    vector<int> a(n+1);
    for(int i=1;i<=n;i++) cin>>a[i];
    
    vector<int> cnt;
    stack<int> sta;
    vector<vector<int>> ans;
    int index = 1;
    while(index <= n) {
        //从盒子中取
        while(sta.size()) {
            if(cnt.size() == k) break;
            if(cnt.empty() || sta.top()<=cnt.back()) {
                cnt.push_back(sta.top());
                sta.pop();
            } else {
                break;
            }
            if(cnt.size() == k) break;
        }
        if(cnt.size() == k) {
            ans.push_back(cnt);
            cnt.clear();
            continue;
        }
        //从推送器取
        while(index <= n) {
            if(cnt.empty() || a[index] <= cnt.back()) {
                cnt.push_back(a[index++]);
            } else {
                if(sta.size() == m) {
                    ans.push_back(cnt);
                    cnt.clear();
                    break;
                } else {
                    sta.push(a[index++]);
                }
            }
            if(cnt.size() == k) {
                ans.push_back(cnt);
                cnt.clear();
                break;
            }
        }
    }
    
    while(sta.size()) {
        while(sta.size() && (cnt.empty() || sta.top()<=cnt.back())) {
            if(cnt.size() == k) break;
            cnt.push_back(sta.top());
            sta.pop();
            if(cnt.size() == k) break;
        }
        if(cnt.size()) {
            ans.push_back(cnt);
            cnt.clear();
        }    
    }
    if(cnt.size()) {
        ans.push_back(cnt);
        cnt.clear();
    }
    
    for(auto its : ans) {
        int len = its.size();
        for(int i=0;i<len;i++) {
            cout<<its[i];
            if(i != len-1) cout<<" ";
        }
        cout<<endl;
    }
}
```
