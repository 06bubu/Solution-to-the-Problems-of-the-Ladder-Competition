void init() {

int n,k,m;
struct node {
    int lose,win;
};

void solve() {
    cin >> k;
    n = (1<<k) - 1;
    vector<node> a(n+1);
    for(int i=n;i>=1;i--) cin>>a[i].lose;
    cin>>a[1].win;

    auto dfs = [&](auto&& dfs,int u) -> bool {
        if(a[u].win < a[u].lose) return 0;
        int L = u<<1;
        int R = u<<1|1;
        if(R > n) return true;

        a[L].win = a[u].win;
        a[R].win = a[u].lose;
        
        if(dfs(dfs,L) && dfs(dfs,R)) return true;
        swap(a[L].win,a[R].win);
        return (dfs(dfs,L)&&dfs(dfs,R));
    };
    if(dfs(dfs,1)) {
        int mini = 1 << (k - 1);
		for (int i = n; i >= mini; i--)
            cout << a[i].win << " " << a[i].lose<<" ";
    } else {
        cout<<"No Solution"<<endl;
    }
}
